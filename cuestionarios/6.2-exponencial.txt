# (I)  

    exponencial <- function(t = 20, lambda = 1.02, N_0 = 10) {
      N <- numeric(t)
      N[1] <- N_0
      for(i in 1:(t - 1)) {
        N[i + 1] <- N[i] * lambda
      }
      N
    }
